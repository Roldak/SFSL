96  virtual CodeGenOutput<T>& operator <<(const T& t) override {
97  if (_here) {
98  _here->next = _ctx->memoryManager().New<Node>(_here->next, t);
99  if (_end == _here) {
100  _end = _end->next;
101  }
102  _here = _here->next;
103  } else if (_end) {
104  _end->next = _ctx->memoryManager().New<Node>(_end->next, t);
105  _here = _end->next;
106  } else {
107  _here = _ctx->memoryManager().New<Node>(nullptr, t);
108  _here->next = _here;
109  _end = _here;
110  }
111  return *this;
112  }

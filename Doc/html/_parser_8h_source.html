89  tok::Token* lastTok = _currentToken;
90 
91  if (!accept(type)) {
92  if (fatal) {
93  _ctx->reporter().fatal(*lastTok, "expected " + expected + "; got " + lastTok->toString());
94  } else {
95  _ctx->reporter().error(*lastTok, "expected " + expected + "; got " + lastTok->toString());
96  }
97  return false;
98  }
99 
100  return true;
